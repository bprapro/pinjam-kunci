function ambilKunci() { const url = "URL_WEB_APP_APPS_SCRIPT_ANDA?aksi=ambil_tersedia"; fetch(url) .then(response => response.json()) .then(data => { const selectKunci = document.getElementById("kunci"); selectKunci.innerHTML = ''; data.forEach(kunci => { const option = document.createElement("option"); option.value = kunci.id; // Menampilkan ID, Nama, Lantai, dan Zona sekaligus di pilihan drop-down option.textContent = `${kunci.id} - ${kunci.nama} [${kunci.lantai} - ${kunci.zona}]`; selectKunci.appendChild(option); }); }) .catch(error => { console.error("Error:", error); alert("Gagal mengambil data kunci tersedia."); }); }
